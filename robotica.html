Algoritmo de montagem

Conecte o GND do arduino aos segmentos negativos da protoboard; Conecte o 5V do arduino aos segmentos positivos da protoboard; Coloque três push button no centro da protoboard; Conecte uma das pernas de todos os push buttons no segmento positivo da protoboard; Conecte a perna oposta a perna alimentada com 5V, ao segmento negativo da protoboard com um resistor de 150Ω, faça isso com todos os push buttons; Conecte a mesma perna da ação anterior de todos os push buttons as respectivas portas digitais do arduino 12, 11 e 10; Coloque o display de 4 digitos na protoboard; Conecte o GND do display ao segmento negativo da protoboard; Conecte o VIN do display ao segmento positivo da protoboard; Conecte o CLK do display a porta analógica 4 do arduino; Conecte o DIO do display a porta analógica 5 do arduino; Coloque um buzzer na protoboard; Conecte o GND do buzzer no segmento negativo da protoboard; Conecte a entrada do buzzer na porta digital 9 do arduino; Coloque um led branco na protoboard; Conecte o ânodo do led ao segmento negativo da protoboard; Conecte um resistor de 150Ω no cátodo do led; Conecte o led no segmento positivo da protoboard(Passando pelo resistor);

#include

#define CLK A4

#define DIO A5

#define Ba 12 #define Bb 11 #define Bc 10 #define Rf 6 #define buzzer 9 int Ba_l = 0; int Bb_l = 0; int Bc_l = 0; int vb =0; TM1637Display display(CLK, DIO); void setup(){ display.setBrightness(15); pinMode(Ba, INPUT); pinMode(Bb, INPUT); pinMode(Bc, INPUT); display.showNumberDec(0, false); noTone(buzzer); Serial.begin(9600); } void loop(){ int Rf_l = digitalRead(Rf); if (Rf_l == 1){ for(int i = 2;i <= 300; i++){ display.showNumberDec(i, false); int Ba_l = digitalRead(Ba); int Bb_l = digitalRead(Bb); int Bc_l = digitalRead(Bc); delay(1); if(Ba_l == 1 ){ vb = 1; break; } if (Bb_l == 1 || Bc_l == 1){ break; } } if (vb == 0){ display.showNumberDec(8888, false); tone(buzzer, 1000, 1000000); delay(1000000); } else{ display.showNumberDec(0, false); noTone(buzzer); vb = 0; } } }
